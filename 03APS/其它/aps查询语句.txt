    a.eqpname = d.eqpname

    lot.pid in

        #{pid}
    </foreach>
and
    d.flag = '0'
and
    c.plant = #{plant}
and
    a.plant = #{shortPlant}

      where specialmachine like '%' || #{custCode} || '%'
   and flag = '0') a

    base_eqpdata b
on
    a.eqpname = b.eqpid

    IC_UPH_T_BIGDATA c

    MES_CIM_EQP_AREA d

  and
      c.device = #{pid}
  and
      b.subplant = #{plant}
  and
      c.plant = #{plant}
  and

select distinct id,
                custCode,
                plant,
                pid,
                bd,
                step,
                sumEqpQty,
                specCount,
                planOutQty
from (select a.id,
             a.cuscode custCode,
             a.plant,
             a.pid,
             a.bd,
             a.step,
             b.sumeqpqty sumEqpQty,
             b.sumplanqty planOutQty,
             case
                 when c.specialmachine is null then
                     0
                 else
                     1
                 end as  specCount
            from
                aps_plan_parent a
            inner join
                aps_plan_parent_detail b
            on
                a.id = b.fromid
            left join
                MES_CIM_EQP_AREA c
            on
                c.specialmachine like '%' || a.cuscode || '%'
            and
                c.flag = '0'
            where
                trunc(a.create_date) = trunc(sysdate)
            <if test="null != plant and plant != ''">
                and
                    a.plant = #{plant}
            </if>
            <if test="null != step and step != ''">
                and
                    a.step = #{step}
            </if>

     )
order by plant desc,specCount desc
